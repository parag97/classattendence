create table user_info(
uid int AUTO_INCREMENT KEY NOT NULL,
fname varchar(36) default 0 not null,
lname varchar(36) default 0 not null,
gender char(4),
dob date not null DEFAULT '2000-01-01',
course varchar(16) not null default 'not'
);
